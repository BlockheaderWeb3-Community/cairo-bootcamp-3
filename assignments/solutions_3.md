fn main() {
   subtract();
   multiply();
   divide(); 
}

fn subtract() -> u16{
    let  a = 10_u16;
    let  b = 20_u16;

   let subt = a - b;
    println!("The answer is: {}" sub);

    sub
}

fn multiply() -> u32 {
    let (a, b): (u32, u32) = (13, 10);

    let mult = a * b;
    println!("The product is: {}", mult);

    mult;
}

fn divide() -> u8{
    let  a: u8 = 14;
    let  b: u8 = 2;

    let div = a / b;
    println!("The answer is: {}" div);

    div;
}
